// 1. USERS (collection principale)
{
  _id: ObjectId,
  nom: String,
  prenom: String,
  email: String (unique),
  password: String (hashé),
  role: String, // "admin", "boutique", "acheteur"
  photo: String, // URL ou base64
  dateCreation: Date,
  actif: Boolean
}

// 2. BOUTIQUES (extension de User)
{
  _id: ObjectId (référence User),
  nomBoutique: String,
  description: String,
  adresse: String,
  telephone: String,
  logo: String,
  modePaiementAcceptes: [String], // ["carte", "espèces", "mobile money"]
  horaires: {
    lundi: { ouverture: "08:00", fermeture: "20:00" },
    // ...
  },
  noteMoyenne: Number,
  totalAvis: Number
}

// 3. ACHETEURS (extension de User)
{
  _id: ObjectId (référence User),
  telephone: String,
  adresseLivraisonParDefaut: {
    rue: String,
    ville: String,
    codePostal: String,
    pays: String
  },
  preferences: [String] // catégories préférées
}

// 4. PRODUITS (catalogue général)
{
  _id: ObjectId,
  nom: String,
  description: String,
  categorie: String, // "alimentaire", "habillement", etc.
  image: String,
  datePeremption: Date (optionnel),
  caracteristiques: Map // flexible
}

// 5. PRODUITS_PAR_BOUTIQUE (offres spécifiques)
{
  _id: ObjectId,
  idBoutique: ObjectId (référence Boutique),
  idProduit: ObjectId (référence Produit),
  prix: Number,
  stock: Number,
  enPromotion: Boolean,
  dateAjout: Date
}

// 6. PROMOTIONS
{
  _id: ObjectId,
  idProduitParBoutique: ObjectId,
  remisePourcentage: Number, // 10, 20, etc.
  dateDebut: Date,
  dateFin: Date,
  actif: Boolean
}

// 7. AVIS
{
  _id: ObjectId,
  idAcheteur: ObjectId,
  idProduitParBoutique: ObjectId,
  note: Number, // 1 à 5
  commentaire: String,
  date: Date,
  reponseBoutique: {
    commentaire: String,
    date: Date
  }
}

// 8. COMMANDES
{
  _id: ObjectId,
  idAcheteur: ObjectId,
  idBoutique: ObjectId,
  dateCommande: Date,
  statut: String, // "en_attente", "confirmée", "préparée", "expédiée", "livrée", "annulée"
  articles: [{
    idProduitParBoutique: ObjectId,
    nomProduit: String,
    prixUnitaire: Number,
    quantite: Number,
    remise: Number
  }],
  adresseLivraison: {
    rue: String,
    ville: String,
    codePostal: String,
    pays: String
  },
  fraisLivraison: Number,
  total: Number,
  modePaiement: String,
  paiementEffectue: Boolean
}

// 9. LIVRAISONS
{
  _id: ObjectId,
  idCommande: ObjectId,
  statut: String, // "en_attente", "en_cours", "livree"
  livreur: String,
  numeroSuivi: String,
  dateEstimee: Date,
  dateLivraison: Date
}

// 10. TARIFS_LIVRAISON (configuration par boutique)
{
  _id: ObjectId,
  idBoutique: ObjectId,
  tarifs: [{
    distanceMin: Number, // en km
    distanceMax: Number,
    prix: Number
  }],
  zoneGratuite: Number // distance offerte
}
